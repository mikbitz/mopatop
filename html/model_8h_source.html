155  void step(int stepNumber, parameterSettings& parameters){
156  //set some timers so loop relative times can be compared - note disease loop tends to get slower as more agents get infected.
157  auto start=timeReporter::getTime();

164  places[i]->update();
165  }
166  if (stepNumber==0){

170  }

172  int infected=0,recovered=0,dead=0;
173  //do disease - synchronous update (i.e. all agents contaminate before getting infected) so that no agent gets to infect ahead of others.

175  #pragma omp parallel for

178  }

181  timeReporter::showInterval("Run time coughing: ",start,end);
182  start=end;
183  }
184  //the disease progresses
185  //This is faster here using an RNG separate for each thread
186  #pragma omp parallel for
187  for (int i=0;i<agents.size();i++){
188  agents[i]->process_disease(randoms[omp_get_thread_num()]);
189  }
190  if (stepNumber==0){
191  end=timeReporter::getTime();
192  timeReporter::showInterval("Run time being diseased: ",start,end);
193  start=end;
194  }

202  }
203  }
