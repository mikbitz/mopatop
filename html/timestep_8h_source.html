120  timeStep(parameterSettings& p){
121  units = p.get("timeStep.units");
122  //make sure the string is all lower case and has no leadin gor trai ling spaces
123  std::for_each(units.begin(), units.end(), [](char & c) {c = std::tolower(c);});
124  units.erase(std::remove_if(units.begin(), units.end(), ::isspace), units.end());
125  //get the base level for the above units
126  dt = p.get<double>("timeStep.dt");
127  //set dt so that units are consistently seconds in this class
128  //this way any other class can get the actual time represented by dt using the access functions below
129  if (units=="years"){ //number of timeSteps in one year = timeStep::years/timeStep::dt, and so on
130  dt *= years;
131  }else if (units=="months"){
132  dt *= months;
133  }else if (units=="days"){
134  dt *= days;
135  }else if (units=="hours"){
136  dt *= hours;
137  }else if (units=="minutes"){
138  dt *= minutes;
139  }else if (units=="seconds"){
140  dt *= seconds;
141  }else {
142  std::cout<<"Invalid time units: "<<units<<" in timeStep.h"<<std::endl;
143  exit(1);
144  }
145  //set the initial date and time using the value from the paramter file in the form "Day dd/mm/yyyy hh:mm:ss"
146  setDate(p.get("timeStep.startdate"));
147  }

172  static void update(){
173  stepNumber++;
174  currentSeconds+=deltaT();//deltaT is always in seconds
175  if (currentSeconds>=60){
176  currentMinute+=currentSeconds/60;
177  currentSeconds=currentSeconds%60;
178  }
179  if (currentMinute>=60){
180  currentHour+=currentMinute/60;
181  currentMinute=currentMinute%60;
182  }
183  if (currentHour>=24){
184  currentWeekDay+=(currentHour/24);
185  currentWeekDay=currentWeekDay%7;
186  currentDayOfMonth+=currentHour/24;
187  currentHour=currentHour%24;
188  }
189  int leapday=0;
190  if (currentDayOfMonth>=monthDays[currentMonth]){
191  if (currentMonth==1){//February, since months run from 0 to 11
192  //leap year if divisible by 4 unless a century in which case needs to be divisible by 400
193  if(currentYear%400==0 || (currentYear%4==0 && currentYear%100!=0))leapday=1;
194  }
195  while(currentDayOfMonth>=monthDays[currentMonth]+leapday){
196  currentDayOfMonth-=monthDays[currentMonth]+leapday;
197  currentMonth++;
198  if (currentMonth!=1) leapday=0;
199  }
200  }
201  if (currentMonth>=12){
202  currentYear+=currentMonth/12;
203  currentMonth=currentMonth%12;
204  }
205  reportDate();
206  }
