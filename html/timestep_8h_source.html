121  timeStep(parameterSettings& p){
122  units = p.get("timeStep.units");
123  //make sure the string is all lower case and has no leadin gor trai ling spaces
124  std::for_each(units.begin(), units.end(), [](char & c) {c = std::tolower(c);});
125  units.erase(std::remove_if(units.begin(), units.end(), ::isspace), units.end());
126  //get the base level for the above units
127  dt = p.get<double>("timeStep.dt");
128  //set dt so that units are consistently seconds in this class
129  //this way any other class can get the actual time represented by dt using the access functions below
130  if (units=="years"){ //number of timeSteps in one year = timeStep::years/timeStep::dt, and so on
131  dt *= years;
132  }else if (units=="months"){
133  dt *= months;
134  }else if (units=="days"){
135  dt *= days;
136  }else if (units=="hours"){
137  dt *= hours;
138  }else if (units=="minutes"){
139  dt *= minutes;
140  }else if (units=="seconds"){
141  dt *= seconds;
142  }else {
143  std::cout<<"Invalid time units: "<<units<<" in timeStep.h"<<std::endl;
144  exit(1);

173  static void update(){
174  stepNumber++;
175  currentSeconds+=deltaT();//deltaT is always in seconds
176  if (currentSeconds>=60){
177  currentMinute+=currentSeconds/60;
178  currentSeconds=currentSeconds%60;
179  }
180  if (currentMinute>=60){
181  currentHour+=currentMinute/60;
182  currentMinute=currentMinute%60;
183  }
184  if (currentHour>=24){
185  currentWeekDay+=(currentHour/24);
186  currentWeekDay=currentWeekDay%7;
187  currentDayOfMonth+=currentHour/24;
188  currentHour=currentHour%24;
189  }
190  int leapday=0;
191  if (currentDayOfMonth>=monthDays[currentMonth]){
192  if (currentMonth==1){//February, since months run from 0 to 11
193  //leap year if divisible by 4 unless a century in which case needs to be divisible by 400
194  if(currentYear%400==0 || (currentYear%4==0 && currentYear%100!=0))leapday=1;
195  }
196  while(currentDayOfMonth>=monthDays[currentMonth]+leapday){
197  currentDayOfMonth-=monthDays[currentMonth]+leapday;
198  currentMonth++;
199  if (currentMonth>=12){
200  currentYear+=currentMonth/12;
201  currentMonth=currentMonth%12;
202  }
203  if (currentMonth!=1)
204  leapday=0;
205  else
206  if(currentYear%400==0 || (currentYear%4==0 && currentYear%100!=0))leapday=1;
207  }
208  }

286  {
287  assert(month>0 && month<=12);
288  int offset[] = {0,3,2,5,0,3,5,1,4,6,2,4};
289  if(month<3) year--;
290  int result= ((year+year/4-year/100+year/400+offset[month-1]+day)%7);//note: relying on integer division truncating any decimals on dividing the year
