90 
91  //set # as comment character - strip all white space from the test string, so that lines can start with spaces followed by #
92  std::string test=line;
93  test.erase(std::remove_if(test.begin(), test.end(), ::isspace), test.end());
94  //need to check end of file again as the above getline may have actually gone past it, if previous read got to the file end
95  if (!infile.eof() && test[ 0 ] != '#'){
96 
97  //separator between parameter name and value is =
98  auto pos = line.find('=');
99 
100  //ignore if no "=" is found - npos is the end of the string
101  if (pos!=std::string::npos){
102  //get string of length pos from position zero - so "=" will not be included in the parameter name
103  label=line.substr(0,pos);
104  //get from place after "=" to the end of the line
105  value=line.substr(pos+1);
106  //set the parameter if it exists
107  if (is_valid(label)){
108  _parameters[label]=value;

110  }

112  }
