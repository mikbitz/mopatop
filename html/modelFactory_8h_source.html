119  void createAgents(parameterSettings& parameters, std::vector<agent*>& agents,std::vector<place*>& places){
120 
121  long nAgents=parameters.get<long>("run.nAgents");
122  //default values to allocate rough numbers of agents to types of place
123  int agentsPerHome=3;
124  int agentsPerWorkPlace=10;
125  int agentsPerBus=30;
126  //excess allows for numbers of agents not strictly divisible by agentsPerHome - create at most one extra place.
127  int excess=((nAgents % agentsPerHome)>0);//if remainder of nAgents on division by agentsPerHome is greater than zero, this will result in 1, otherwise zero
128  long nHomes=nAgents/agentsPerHome+excess;
129  //now get number of workplaces
130  excess=((nAgents % agentsPerWorkPlace)>0);//as with homes, allow for not divisible by 10
131  int nWork=nAgents/agentsPerWorkPlace+excess;
132  //and number of transport vehciles, assumed buses
133  excess=((nAgents % agentsPerBus)>0);//allow for not exactly 30 agents per bus
134  long nBus=nAgents/agentsPerBus+excess;
135 
136  //faster to resize the array here, then create places in a parallel loop (individual place memory allocations get done in parallel, but loop is thread-safe)
137  //not sure if one could also subdivide this resizing and gain something in speed by making indiviual sub-vectors on each thread in an omp parallel loop then concatenating them in an omp critical section?
138  //Currently 4.5e9 agents + 2.1 e9 places on 64 threads on HPC takes about 45 minutes.
139  places.resize(nHomes+nWork+nBus);
140 
141  std::cout<<"Starting simple mobile generator..."<<std::endl;
142  std::cout<<"Creating homes ..."<<std::endl;
143 
144  #pragma omp parallel for
145  for (long i=0;i<nHomes;i++){
146  place* p=new place(parameters);
147  p->setID(i);
148  places[i]=p;
149  }
150 
151  std::cout<<"Creating agents ...";
152  long k=0;
153  //fraction indicates when each extra 10% of agents have been created
154  long fr=parameters.get<long>("run.nAgents")/10;
155 
156  //allocate agentsPerHome agents per home - as far as possible - any excess over nAgents/agentsPerHome go into the excess Home as defined above (either one or two if agentsPerHome==3 for example)
157  //again fastest to resize vector first, then allocate agent pointers in parallel loop.
158  agents.resize(nAgents);
159  #pragma omp parallel for
160  for (long i=0;i<nAgents;i++){
161  agent* a=new agent();
162  a->setHome(places[i/agentsPerHome]);
163  k++;
164  if (k%fr==0)std::cout<<k<<"...";
165  agents[i]=a;
166  }
167  std::cout<<std::endl;
168  //create work places - (nAgents / agentsPerWorkPlace) as many as agents - add them on to the end of the place list.
169  std::cout<<"Creating workplaces ..."<<std::endl;
170 
171  #pragma omp parallel for
172  for (long i=nHomes;i<nHomes+nWork;i++){
173  place* p=new place(parameters);
174  p->setID(i);
175  places[i]=p;
176  }
177  //shuffle agents so household members get different workplaces - can this be parallelised?
178  random_shuffle(agents.begin(),agents.end());
179  //allocate agentsPerWorkPlace agents per workplace
180  #pragma omp parallel for
181  for (long i=0;i<agents.size();i++){
182  assert(places[i/agentsPerWorkPlace+nHomes]!=0);
183  agents[i]->setWork(places[i/agentsPerWorkPlace+nHomes]);
184  }
185  std::cout<<"Creating transport ..."<<std::endl;
186  //create buses - (nAgents / agentsPerBus) since agentsPerBus agents per bus. add them to the end of the place list again
187 
188  #pragma omp parallel for
189  for (long i=nHomes+nWork;i<nHomes+nWork+nBus;i++){
190  place* p=new place(parameters);
191  p->setID(i);
192  places[i]=p;
193  }
194  //allocate agentsPerBus agents per bus - since agents aren't shuffled again, those in similar workplaces will tend to share buses.
195  #pragma omp parallel for
196  for (long i=0;i<agents.size();i++){
197  assert(places[i/agentsPerBus+nHomes+nWork]!=0);
198  agents[i]->setTransport(places[i/agentsPerBus+nHomes+nWork]);
199  }
200  //set up travel schedule - same for every agent at the moment - so agents are all on the bus, at work or at home at exactly the same times
201  #pragma omp parallel for
202  for (long i=0;i<agents.size();i++){
203  agents[i]->initTravelSchedule(parameters);
204  }
205  //report intialization to std out
206  std::cout<<"Built "<<agents.size()<<" agents and "<<places.size()<<" places."<<std::endl;
207 
208  }
